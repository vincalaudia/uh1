Two Pointer

Use two indices moving toward each other or same direction. O(n) time, O(1) space. 

def two_sum_sorted(arr, target):
    l, r = 0, len(arr) - 1
    while l < r:
        s = arr[l] + arr[r]
        if   s == target: return [l, r]
        elif s <  target: l += 1
        else:             r -= 1
    return []

Sliding Window

pattern

Expand right, shrink left. Longest substring without repeating characters. 

def longest_no_repeat(s):
    seen = {}
    l = best = 0
    for r, c in enumerate(s):
        if c in seen and seen[c] >= l:
            l = seen[c] + 1
        seen[c] = r
        best = max(best, r - l + 1)
    return best

Output
3 (window "abc")

Kadane's Algorithm — Maximum Subarray

classic 
def max_subarray(nums):
    cur = best = nums[0]
    for n in nums[1:]:
        cur = max(n, cur + n)   # extend or restart?
        best = max(best, cur)
    return best

// Case Study
Spotify: Finding Duplicate Songs
Spotify needs to detect duplicate songs in a playlist. They store songs as arrays of audio fingerprint values. The problem reduces to: "find all duplicates in an array." Using a frequency counter (hash map over the array) gives O(n) time vs O(n²) brute force. Arrays enable fast sequential processing of audio data. 
def find_duplicates(nums):
    # Use indices as hash — for arrays with values 1..n
    result = []
    for n in nums:
        idx = abs(n) - 1
        if nums[idx] < 0:
            result.append(abs(n))
        else:
            nums[idx] = -nums[idx]
    return result
# Input:  [4,3,2,7,8,2,3,1]
# Output: [2,3]   O(n) time, O(1) extra space

Reverse Linked List

def reverse(head):
    prev = None
    curr = head
    while curr:
        nxt       = curr.next
        curr.next = prev   # flip!
        prev      = curr
        curr      = nxt
    return prev            # new head

# 1→2→3→None  becomes  3→2→1→None

Floyd's Cycle Detection

must know

Slow moves 1 step, fast moves 2. If cycle exists, they meet.

def has_cycle(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
        if slow == fast:
            return True
    return False

# Find middle (same trick):
def find_mid(head):
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
    return slow
💡 Golden Tip Always use a dummy head node when modifying linked lists. It eliminates edge cases where the head itself is deleted or changed.
dummy = ListNode(0); dummy.next = head

Problem 2: Monotonic Stack — Next Greater Element

pattern

Keep the stack in monotonically decreasing order. When a larger element is found, it's the "next greater" for all smaller elements in the stack. 

def next_greater(nums):
    res = [-1] * len(nums)
    stack = []   # stores indices (monotonic decreasing values)
    for i, n in enumerate(nums):
        while stack and nums[stack[-1]] < n:
            res[stack.pop()] = n
        stack.append(i)
    return res

[2, 1, 2, 4, 3]
Output
[4, 2, 4, -1, -1]

Problem 3: Min Stack

design 
class MinStack:
    def __init__(self):
        self.stack = []
        self.min_stack = []  # tracks minimums

    def push(self, val):
        self.stack.append(val)
        m = val if not self.min_stack else min(val, self.min_stack[-1])
        self.min_stack.append(m)

    def pop(self):
        self.stack.pop(); self.min_stack.pop()

    def get_min(self): return self.min_stack[-1]  # O(1)!

BFS with Queue — Shortest Path

classic 
from collections import deque

def bfs_shortest(graph, start, end):
    visited = {start}
    q = deque([(start, [start])])  # (node, path)
    while q:
        node, path = q.popleft()
        if node == end: return path
        for nb in graph.get(node, []):
            if nb not in visited:
                visited.add(nb)
                q.append((nb, path + [nb]))
    return None

Level-Order Tree Traversal

classic 
from collections import deque

def level_order(root):
    if not root: return []
    q, res = deque([root]), []
    while q:
        level = []
        for _ in range(len(q)):  # snapshot current level size
            node = q.popleft()
            level.append(node.val)
            if node.left:  q.append(node.left)
            if node.right: q.append(node.right)
        res.append(level)
    return res

Classic: Two Sum — O(n)

must know 
def two_sum(nums, target):
    seen = {}  # value → index
    for i, n in enumerate(nums):
        complement = target - n
        if complement in seen:
            return [seen[complement], i]
        seen[n] = i
    return []

The 4 Traversals

must memorize 
# INORDER:   Left → Root → Right (BST = sorted)
def inorder(root):
    if not root: return []
    return inorder(root.left)+[root.val]+inorder(root.right)

# PREORDER:  Root → Left → Right (copy tree)
def preorder(root):
    if not root: return []
    return [root.val]+preorder(root.left)+preorder(root.right)

# POSTORDER: Left → Right → Root (delete tree)
def postorder(root):
    if not root: return []
    return postorder(root.left)+postorder(root.right)+[root.val]

# LEVEL-ORDER: BFS with queue
from collections import deque
def level_order(root):
    if not root: return []
    q, res = deque([root]), []
    while q:
        node = q.popleft(); res.append(node.val)
        if node.left:  q.append(node.left)
        if node.right: q.append(node.right)
    return res

Max Depth & LCA

classic 
def max_depth(root):
    if not root: return 0
    return 1 + max(max_depth(root.left),
                     max_depth(root.right))

def lca(root, p, q):
    if not root or root==p or root==q:
        return root
    left  = lca(root.left,  p, q)
    right = lca(root.right, p, q)
    return root if left and right else left or right

Validate BST

classic 
def is_valid_bst(root,
    lo=float('-inf'), hi=float('inf')):
    if not root: return True
    if not (lo < root.val < hi): return False
    return (
        is_valid_bst(root.left,  lo, root.val) and
        is_valid_bst(root.right, root.val, hi)
    )

BFS vs DFS

comparison 
from collections import deque

# ── BFS — uses QUEUE ────────────────────
def bfs(graph, start):
    visited = {start}
    q = deque([start])
    while q:
        node = q.popleft()
        for nb in graph[node]:
            if nb not in visited:
                visited.add(nb)
                q.append(nb)

# ── DFS — uses STACK or recursion ───────
def dfs(graph, node, visited=None):
    if visited is None: visited = set()
    visited.add(node)
    for nb in graph[node]:
        if nb not in visited:
            dfs(graph, nb, visited)
    return visited
BFS vs DFS Rule BFS = shortest path, level-by-level, uses Queue
DFS = explore fully, cycle detect, uses Stack/Recursion
Grid problems → DFS usually simpler

Number of Islands — Grid DFS

must know 
def num_islands(grid):
    if not grid: return 0
    rows, cols, count = len(grid), len(grid[0]), 0

    def dfs(r, c):
        if r < 0 or r >= rows or c < 0 or c >= cols or grid[r][c] != '1': return
        grid[r][c] = '0'  # mark visited
        for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]:
            dfs(r+dr, c+dc)

    for r in range(rows):
        for c in range(cols):
            if grid[r][c] == '1': dfs(r, c); count += 1
    return count

def count_components(n, edges):
    graph = defaultdict(list)
    for u, v in edges:
        graph[u].append(v); graph[v].append(u)
    visited, count = set(), 0
    def dfs(node):
        visited.add(node)
        for nb in graph[node]:
            if nb not in visited: dfs(nb)
    for i in range(n):
        if i not in visited: dfs(i); count += 1
    return count

Merge Sort

stable, guaranteed O(n log n) 
def merge_sort(arr):
    if len(arr) <= 1: return arr
    mid   = len(arr) // 2
    left  = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])
    return _merge(left, right)

def _merge(l, r):
    res, i, j = [], 0, 0
    while i < len(l) and j < len(r):
        if l[i] <= r[j]: res.append(l[i]); i += 1
        else:             res.append(r[j]); j += 1
    return res + l[i:] + r[j:]

Core Template

memorize 
def binary_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if   arr[mid] == target: return mid
        elif arr[mid] <  target: left  = mid + 1
        else:                    right = mid - 1
    return -1  # not found

Find First & Last Position

variant 
def find_first(arr, target):
    l, r, res = 0, len(arr)-1, -1
    while l <= r:
        m = (l + r) // 2
        if arr[m] == target: res = m; r = m - 1
        elif arr[m] < target: l = m + 1
        else: r = m - 1
    return res

Search in Rotated Sorted Array

hard variant 
def search_rotated(nums, target):
    l, r = 0, len(nums) - 1
    while l <= r:
        m = (l + r) // 2
        if nums[m] == target: return m
        if nums[l] <= nums[m]:          # left half is sorted
            if nums[l] <= target < nums[m]: r = m - 1
            else: l = m + 1
        else:                           # right half is sorted
            if nums[m] < target <= nums[r]: l = m + 1
            else: r = m - 1
    return -1

Dynamic Programming

The Two Approaches

concept 
# ── TOP-DOWN (Memoization) ───────────────
from functools import lru_cache

@lru_cache(maxsize=None)
def fib(n):
    if n <= 1: return n
    return fib(n-1) + fib(n-2)
# Time: O(n)   Space: O(n)

# ── BOTTOM-UP (Tabulation) ───────────────
def fib_bu(n):
    if n <= 1: return n
    dp = [0] * (n+1)
    dp[1] = 1
    for i in range(2, n+1):
        dp[i] = dp[i-1] + dp[i-2]
    return dp[n]

# Space-optimized O(1):
def fib_opt(n):
    a, b = 0, 1
    for _ in range(n): a, b = b, a+b
    return a

def climb_stairs(n):
    if n <= 2: return n
    a, b = 1, 2
    for _ in range(3, n+1):
        a, b = b, a + b
    return b

Coin Change — Bottom-Up DP

classic 
def coin_change(coins, amount):
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0   # base: 0 coins for amount 0
    for a in range(1, amount + 1):
        for c in coins:
            if c <= a:
                dp[a] = min(dp[a], dp[a - c] + 1)
    return dp[amount] if dp[amount] != float('inf') else -1
Input
coins=[1,5,6,9], amount=11
Output
2 (5+6=11)

Longest Common Subsequence (LCS) — 2D DP

classic 2D 
def lcs(s1, s2):
    m, n = len(s1), len(s2)
    dp = [[0]*(n+1) for _ in range(m+1)]
    for i in range(1, m+1):
        for j in range(1, n+1):
            if s1[i-1] == s2[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])
    return dp[m][n]

s1="ABCBDAB"
s2="BDCAB"
Output
4
("BCAB")

Autocorrect: Edit Distance (Levenshtein)
When you type "helo" and your phone suggests "hello", it's computing the edit distance — minimum insertions, deletions, substitutions to transform one string to another. This is a 2D DP problem (dp[i][j] = edit distance of s1[:i] and s2[:j]). Every major NLP system uses this under the hood. 
def min_distance(w1, w2):
    m, n = len(w1), len(w2)
    dp = [[i+j if i*j==0 else 0 for j in range(n+1)] for i in range(m+1)]
    for i in range(1, m+1):
        for j in range(1, n+1):
            if w1[i-1]==w2[j-1]: dp[i][j] = dp[i-1][j-1]
            else: dp[i][j] = 1 + min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1])
    return dp[m][n]

Top Tips
01

Use x in set, never x in list, for repeated lookups

x in list is O(n) — it scans every element. x in set is O(1). If you need repeated membership checks, convert to a set first: seen = set(nums).

02

Use deque not list for queue operations

list.pop(0) is O(n) because it shifts all elements. deque.popleft() is O(1). This matters in BFS — using a list as a queue makes BFS O(n²) instead of O(V+E).

03

Use @lru_cache to instantly memoize any recursive function

from functools import lru_cache then decorate with @lru_cache(maxsize=None). Turns any naive O(2^n) recursion into O(n) automatically. Only works on hashable arguments (no lists — use tuples).

04

Tuple unpacking for clean swaps and multi-returns

a, b = b, a swaps without a temp variable. Functions can return multiple values: return lo, hi and unpack with lo, hi = func(). Also: first, *rest = arr unpacks head and tail.

05

float('inf') and float('-inf') as sentinels

Use float('inf') to initialize "minimum so far" tracking, or as the default in DP tables (impossible state). Python integers don't overflow, but inf compares correctly with all numbers.

06

Walrus operator := for loop conditions (Python 3.8+)

while chunk := f.read(8192): or if m := pattern.search(text):. Assigns and evaluates in one step. Useful in BFS/DFS when checking and storing simultaneously.
